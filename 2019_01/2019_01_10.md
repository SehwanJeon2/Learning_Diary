def solution(a, b):
    gcd = 0
    lcm = 0
    
    if a > b:
        gcd_a = a
        gcd_b = b
    else:
        gcd_a = b
        gcd_b = a
        
    while gcd_a % gcd_b != 0:
        temp = gcd_a - gcd_b
        if temp > gcd_b:
            gcd_a = temp
        else:
            gcd_a = gcd_b
            gcd_b = temp
    gcd = gcd_b
    
    lcm = int(a*b/gcd)
    
    result = [gcd, lcm]
    
    return result
